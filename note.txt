#Buat Tablespace Temp baru
SQL> CREATE TEMPORARY TABLESPACE temp2
   2  TEMPFILE '/oracle/P04/sapdata4/temp_1/temp.data2' SIZE 50M REUSE
   3  AUTOEXTEND ON NEXT 5M MAXSIZE unlimited
   4  EXTENT MANAGEMENT LOCAL UNIFORM SIZE 10M;

#Set Tablespace Temp baru jadi default
SQL> ALTER DATABASE DEFAULT TEMPORARY TABLESPACE temp2;

#Hapus Tablespace Temp lama
SQL> DROP TABLESPACE PSAPTEMP INCLUDING CONTENTS AND DATAFILES;

#Buat Tablespace PSAPTEMP baru
SQL> CREATE TEMPORARY TABLESPACE PSAPTEMP2
   2  TEMPFILE '/oracle/P04/sapdata4/temp_1/newtemp.temp1' SIZE 1000M REUSE
   3  AUTOEXTEND ON NEXT 100M MAXSIZE unlimited
   4  EXTENT MANAGEMENT LOCAL UNIFORM SIZE 10M;


#Set Tablespace PSAPTEMP2 yang baru jadi default
SQL> ALTER DATABASE DEFAULT TEMPORARY TABLESPACE PSAPTEMP2;
